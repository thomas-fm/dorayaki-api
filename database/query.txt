CREATE TABLE `variant` (
  `ID_Variant` Int(8) AUTO_INCREMENT NOT NULL,
  `Flavour` VarChar(255) CHARACTER SET utf8 NOT NULL,
  `Description` VarChar(1000) CHARACTER SET utf8 NOT NULL,
  `IMG_URL` VarChar(500) CHARACTER SET utf8 NOT NULL,
  CONSTRAINT `unique_id` UNIQUE( `ID_Variant` )
)

CREATE TABLE `store` (
  `ID_Store` Int(8) AUTO_INCREMENT NOT NULL,
  `Name` VarChar(255) CHARACTER SET utf8 NOT NULL,
  `Street` VarChar(255) CHARACTER SET utf8 NOT NULL,
  `District` VarChar(255) CHARACTER SET utf8 NOT NULL,
  `Province` VarChar(255) CHARACTER SET utf8 NOT NULL,
  `Phone_Number` CHAR(16) CHARACTER SET utf8 NOT NULL,
  CONSTRAINT `unique_id` UNIQUE( `ID_Store` )
)

CREATE TABLE `stock` (
  `ID_Store` Int(8) NOT NULL,
  `ID_Variant` Int(8) NOT NULL,
  `Total` Int(8) NOT NULL,
  FOREIGN KEY (`ID_Store`) REFERENCES store(`ID_Store`),
  FOREIGN KEY (`ID_Variant`) REFERENCES variant(`ID_Variant`)
)
